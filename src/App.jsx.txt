import React, { useState, useRef, useEffect } from 'react';

/* Environment variable key, if provided by the platform */
const envApiKey = ""; 

/* Pre-defined demo structural dataset to render inside our CAD Canvas Engine */
const DEMO_STRUCTURE = {
  columns: [
    { x: 150, y: 120, label: 'C1 (300x450)', load: '450kN' },
    { x: 450, y: 120, label: 'C2 (300x450)', load: '480kN' },
    { x: 150, y: 380, label: 'C3 (300x450)', load: '520kN' },
    { x: 450, y: 380, label: 'C4 (300x450)', load: '550kN' }
  ],
  beams: [
    { fromX: 150, fromY: 120, toX: 450, toY: 120, label: 'B1 (250x600)' },
    { fromX: 150, fromY: 380, toX: 450, toY: 380, label: 'B2 (250x600)' },
    { fromX: 150, fromY: 120, toX: 150, toY: 380, label: 'B3 (250x500)' },
    { fromX: 450, fromY: 120, toX: 450, toY: 380, label: 'B4 (250x500)' }
  ],
  dimensions: [
    { fromX: 150, fromY: 80, toX: 450, toY: 80, label: 'Span: 6.00m' },
    { fromX: 90, fromY: 120, toX: 90, toY: 380, label: 'Span: 5.20m' }
  ]
};

export default function App() {
  /* State Management */
  const [apiKey, setApiKey] = useState(envApiKey);
  const [showApiInput, setShowApiInput] = useState(false);
  const [documents, setDocuments] = useState([]);
  const [selectedDocId, setSelectedDocId] = useState(null);
  
  /* Custom toast notifications (No native alert() used) */
  const [toast, setToast] = useState({ show: false, message: '', type: 'info' });

  /* Chat interface states */
  const [chatMessages, setChatMessages] = useState([
    { role: 'model', text: 'Welcome to EZ Cad AI! 🏗️ I am connected to Gemini 2.5 Flash.\n\nPlease input your Gemini API Key in Settings to get started, then try our Demo CAD blueprint or upload your own files.' }
  ]);
  const [chatInput, setChatInput] = useState('');
  const [isTyping, setIsTyping] = useState(false);

  /* CAD Interactive Viewport States */
  const [zoom, setZoom] = useState(1);
  const [pan, setPan] = useState({ x: 0, y: 0 });
  const [isDragging, setIsDragging] = useState(false);
  const [dragStart, setDragStart] = useState({ x: 0, y: 0 });
  
  /* CAD Layers */
  const [showGrid, setShowGrid] = useState(true);
  const [showLoadVectors, setShowLoadVectors] = useState(true);
  const [showStressHeatmap, setShowStressHeatmap] = useState(false);

  /* Specialized AI Action States */
  const [activeInsightTab, setActiveInsightTab] = useState('audit'); 
  const [insightOutput, setInsightOutput] = useState({
    audit: "Upload or select a CAD drawing, then click '✨ Analyze Load Paths' to generate a structural integrity audit.",
    boq: "Upload or select a CAD drawing, then click '✨ Estimate Bill of Quantities' to draft a materials estimation.",
    rfi: "Upload or select a CAD drawing, then click '✨ Draft Technical RFI' to compose client/architect queries."
  });
  const [isInsightLoading, setIsInsightLoading] = useState(false);

  const fileInputRef = useRef(null);
  const chatEndRef = useRef(null);
  const viewportRef = useRef(null);

  const isApiConnected = apiKey.trim().length > 10;

  /* Auto-scroll chat to bottom */
  useEffect(() => {
    chatEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  }, [chatMessages]);

  const showToastMessage = (message, type = 'info') => {
    setToast({ show: true, message, type });
    setTimeout(() => setToast({ show: false, message: '', type: 'info' }), 4000);
  };

  /* Helper to load standard structural layout for immediate testing */
  const loadDemoDocument = () => {
    const demoDoc = {
      id: 'demo-blueprint',
      name: '✨ DEMO_Frame_Slab_Layout.pdf',
      type: 'application/pdf',
      dataUrl: '',
      base64: '',
      isDemo: true
    };
    setDocuments(prev => {
      if (prev.some(d => d.id === 'demo-blueprint')) {
        showToastMessage("Demo layout is already in your workspace!", "warning");
        return prev;
      }
      return [...prev, demoDoc];
    });
    setSelectedDocId('demo-blueprint');
    showToastMessage("Demo CAD vector layout loaded in the viewport!", "success");
  };

  /* Handle File Uploads */
  const handleFileUpload = (e) => {
    const files = Array.from(e.target.files);
    
    files.forEach(file => {
      const reader = new FileReader();
      reader.onload = (event) => {
        const dataUrl = event.target.result;
        const base64 = dataUrl.split(',')[1];
        
        const newDoc = {
          id: Math.random().toString(36).substring(7),
          name: file.name,
          type: file.type,
          dataUrl: dataUrl,
          base64: base64
        };
        
        setDocuments(prev => [...prev, newDoc]);
        if (!selectedDocId) setSelectedDocId(newDoc.id);
        showToastMessage(`Successfully uploaded "${file.name}"`, "success");
      };
      reader.readAsDataURL(file);
    });
    
    if (fileInputRef.current) fileInputRef.current.value = '';
  };

  const removeDocument = (e, id) => {
    e.stopPropagation();
    const filtered = documents.filter(doc => doc.id !== id);
    setDocuments(filtered);
    if (selectedDocId === id) {
      setSelectedDocId(filtered.length > 0 ? filtered[0].id : null);
    }
    showToastMessage("Document removed", "info");
  };

  /* CAD Navigation handlers */
  const handleMouseDown = (e) => {
    if (e.button !== 0) return; 
    setIsDragging(true);
    setDragStart({ x: e.clientX - pan.x, y: e.clientY - pan.y });
  };

  const handleMouseMove = (e) => {
    if (!isDragging) return;
    setPan({
      x: e.clientX - dragStart.x,
      y: e.clientY - dragStart.y
    });
  };

  const handleMouseUp = () => {
    setIsDragging(false);
  };

  const handleZoom = (factor) => {
    setZoom(prev => Math.max(0.3, Math.min(prev * factor, 4)));
  };

  const resetView = () => {
    setZoom(1);
    setPan({ x: 0, y: 0 });
    showToastMessage("Viewport view reset", "info");
  };

  /* Generalized helper for API requests utilizing gemini-2.5-flash */
  const callGeminiLLM = async (promptText, docList = []) => {
    const documentParts = docList
      .filter(doc => doc.base64)
      .map(doc => ({
        inlineData: {
          mimeType: doc.type,
          data: doc.base64
        }
      }));

    const payload = {
      contents: [
        {
          role: "user",
          parts: [
            ...documentParts,
            { text: promptText }
          ]
        }
      ]
    };

    const modelUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${apiKey}`;

    const response = await fetch(modelUrl, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload)
    });

    const data = await response.json();
    if (data.error) {
      throw new Error(data.error.message || "API error occurred");
    }

    return data.candidates?.[0]?.content?.parts?.[0]?.text || "No output returned.";
  };

  /* Specialized Feature Action Callbacks */
  const handleRunAudit = async () => {
    if (!isApiConnected) {
      showToastMessage("Please enter your Gemini API key in Settings first.", "warning");
      setShowApiInput(true);
      return;
    }
    const targetDoc = documents.find(d => d.id === selectedDocId);
    if (!targetDoc) {
      showToastMessage("Please select or upload a document first.", "warning");
      return;
    }

    setIsInsightLoading(true);
    try {
      const prompt = `You are a Senior Principal Structural Engineer. Analyze the attached document called (${targetDoc.name}). Provide a comprehensive "Structural & Civil Engineering Audit Report". Include professional remarks on: 1. Load Path Analysis (e.g. beam-to-column transfers, column schedules) 2. Reinforcement compliance & shear reinforcement flags 3. Critical structural vulnerabilities. Use structured Markdown with clear headings and bullet points.`;
      const resultText = await callGeminiLLM(prompt, [targetDoc]);
      setInsightOutput(prev => ({ ...prev, audit: resultText }));
      showToastMessage("Structural Integrity Audit Completed! ✨", "success");
    } catch (err) {
      showToastMessage(`Audit Failed: ${err.message}`, "error");
    } finally {
      setIsInsightLoading(false);
    }
  };

  const handleEstimateBoQ = async () => {
    if (!isApiConnected) {
      showToastMessage("Please enter your Gemini API key in Settings first.", "warning");
      setShowApiInput(true);
      return;
    }
    const targetDoc = documents.find(d => d.id === selectedDocId);
    if (!targetDoc) {
      showToastMessage("Please select or upload a document first.", "warning");
      return;
    }

    setIsInsightLoading(true);
    try {
      const prompt = `You are an expert structural quantity surveyor. Scan the attached drawing/spec sheet (${targetDoc.name}). Provide a "Draft Bill of Quantities (BoQ) & Material Estimation Summary". Estimate or extract: 1. Reinforcement Steel Bars (Rebar tonnage estimate based on standard design). 2. Concrete Class Volume (e.g., C30/37 volume estimate for slab or footings). 3. Expected Formwork surface areas. Format this nicely using Markdown tables or lists so it reads like an official engineering estimate document.`;
      const resultText = await callGeminiLLM(prompt, [targetDoc]);
      setInsightOutput(prev => ({ ...prev, boq: resultText }));
      showToastMessage("Material Estimation Generated! ✨", "success");
    } catch (err) {
      showToastMessage(`BoQ Estimation Failed: ${err.message}`, "error");
    } finally {
      setIsInsightLoading(false);
    }
  };

  const handleDraftRFI = async () => {
    if (!isApiConnected) {
      showToastMessage("Please enter your Gemini API key in Settings first.", "warning");
      setShowApiInput(true);
      return;
    }
    const targetDoc = documents.find(d => d.id === selectedDocId);
    if (!targetDoc) {
      showToastMessage("Please select or upload a document first.", "warning");
      return;
    }

    setIsInsightLoading(true);
    try {
      const prompt = `You are a structural project engineer reviewing drafting submittals (${targetDoc.name}). Review the layouts and generate a professional formal technical RFI (Request For Information) draft. State clearly: - RFI Title & Reference - The ambiguity, conflict, or lack of clarity found in the detail. - Proposed solution or engineering clarification recommendation. Format in a highly official submittal letter style ready to be copied into Procore or Aconex.`;
      const resultText = await callGeminiLLM(prompt, [targetDoc]);
      setInsightOutput(prev => ({ ...prev, rfi: resultText }));
      showToastMessage("Technical RFI Draft Created! ✨", "success");
    } catch (err) {
      showToastMessage(`RFI Drafting Failed: ${err.message}`, "error");
    } finally {
      setIsInsightLoading(false);
    }
  };

  /* Standard Chat Prompt Call using gemini-2.5-flash */
  const handleSendMessage = async () => {
    if (!chatInput.trim()) return;
    if (!isApiConnected) {
      showToastMessage("Please connect your Gemini API key in Settings first.", "warning");
      setShowApiInput(true);
      return;
    }

    const userMessage = chatInput;
    setChatInput('');
    setChatMessages(prev => [...prev, { role: 'user', text: userMessage }]);
    setIsTyping(true);

    try {
      const activeDoc = documents.find(d => d.id === selectedDocId);
      const docListToSend = activeDoc ? [activeDoc] : [];

      const contextText = documents.length > 0 
        ? `You are an AI Structural Engineering Expert. Analyze the attached drawing details to answer user inquiries about: ${activeDoc ? activeDoc.name : 'all loaded sheets'}.` 
        : "You are an AI Structural Engineering Expert assistant.";

      const replyText = await callGeminiLLM(`${contextText}\n\nUser Question: ${userMessage}`, docListToSend);
      setChatMessages(prev => [...prev, { role: 'model', text: replyText }]);

    } catch (error) {
      setChatMessages(prev => [...prev, { role: 'model', text: `Error: ${error.message}. Please check your Gemini API connection.` }]);
    } finally {
      setIsTyping(false);
    }
  };

  const selectedDoc = documents.find(d => d.id === selectedDocId);

  return (
    <div className="flex h-screen w-full bg-gradient-to-br from-blue-100 via-indigo-50 to-pink-100 font-sans overflow-hidden text-gray-800 relative select-none">
      
      {/* --- INLINE TOAST SYSTEM --- */}
      {toast.show && (
        <div className="absolute top-6 left-1/2 transform -translate-x-1/2 z-[100] px-6 py-3 rounded-full shadow-lg border backdrop-blur-md animate-bounce flex items-center gap-2 text-sm font-semibold transition-all duration-300 bg-white/90 text-slate-800 border-white/80">
          <span className={
            toast.type === 'success' ? 'text-green-500' :
            toast.type === 'error' ? 'text-red-500' :
            toast.type === 'warning' ? 'text-amber-500' : 'text-blue-500'
          }>●</span>
          {toast.message}
        </div>
      )}

      {/* --- TOP RIGHT NAV: API SETTINGS --- */}
      <div className="absolute top-4 right-4 z-50 flex flex-col items-end">
        <div className="flex items-center gap-3 bg-white/40 backdrop-blur-md border border-white/60 p-2 rounded-full shadow-sm">
          <div className="flex items-center gap-2 px-2">
            <span className={`w-3 h-3 rounded-full transition-all duration-300 ${isApiConnected ? 'bg-green-500 shadow-[0_0_8px_rgba(34,197,94,0.6)]' : 'bg-red-500 shadow-[0_0_8px_rgba(239,68,68,0.6)]'}`}></span>
            <span className="text-sm font-semibold text-gray-700 hidden sm:block">
              {isApiConnected ? 'Gemini 2.5 Flash: Connected' : 'Gemini Disconnected'}
            </span>
          </div>
          <button 
            onClick={() => setShowApiInput(!showApiInput)}
            className="bg-white/60 hover:bg-white/80 text-blue-600 border border-blue-200/50 text-sm font-medium px-4 py-1.5 rounded-full transition flex items-center gap-2"
          >
            <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M10.325 4.317c.426-1.756 2.924-1.756 3.35 0a1.724 1.724 0 002.573 1.066c1.543-.94 3.31.826 2.37 2.37a1.724 1.724 0 001.065 2.572c1.756.426 1.756 2.924 0 3.35a1.724 1.724 0 00-1.066 2.573c.94 1.543-.826 3.31-2.37 2.37a1.724 1.724 0 00-2.572 1.065c-.426 1.756-2.924 1.756-3.35 0a1.724 1.724 0 00-2.573-1.066c-1.543.94-3.31-.826-2.37-2.37a1.724 1.724 0 00-1.065-2.572c-1.756-.426-1.756-2.924 0-3.35a1.724 1.724 0 001.066-2.573c-.94-1.543.826-3.31 2.37-2.37.996.608 2.296.07 2.572-1.065z" /><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M15 12a3 3 0 11-6 0 3 3 0 016 0z" /></svg>
            API Settings
          </button>
        </div>

        {/* API Key Modal */}
        {showApiInput && (
          <div className="mt-2 p-5 bg-white/70 backdrop-blur-xl border border-white/60 rounded-2xl shadow-xl w-80 animate-in fade-in slide-in-from-top-2">
            <h3 className="block text-sm font-bold text-gray-800 mb-1">Gemini 2.5 Flash API Key</h3>
            <p className="text-xs text-gray-500 mb-3">Your API Key is saved locally in memory.</p>
            <input 
              type="password" 
              value={apiKey}
              onChange={(e) => setApiKey(e.target.value)}
              placeholder="AIzaSy..."
              className="w-full p-2.5 mb-3 border border-gray-300/50 rounded-xl bg-white/60 focus:outline-none focus:ring-2 focus:ring-blue-400 text-sm shadow-inner"
            />
            <button 
              onClick={() => setShowApiInput(false)}
              className="w-full bg-gradient-to-r from-blue-500 to-pink-500 hover:from-blue-600 hover:to-pink-600 text-white font-semibold py-2 rounded-xl transition shadow-md"
            >
              Connect API
            </button>
          </div>
        )}
      </div>

      {/* --- LEFT SIDEBAR: DOCUMENT MANAGEMENT --- */}
      <div className="w-72 m-4 mr-2 bg-white/40 backdrop-blur-xl border border-white/60 rounded-3xl flex flex-col shadow-lg p-5 z-10">
        <div className="flex items-center gap-2 mb-8 mt-2">
          <div className="bg-gradient-to-br from-blue-500 to-pink-500 text-white p-2 rounded-xl shadow-md">
            <svg className="w-6 h-6" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M19 11H5m14 0a2 2 0 012 2v6a2 2 0 01-2 2H5a2 2 0 01-2-2v-6a2 2 0 012-2m14 0V9a2 2 0 00-2-2M5 11V9a2 2 0 002-2m0 0V5a2 2 0 012-2h6a2 2 0 012 2v2M7 7h10" /></svg>
          </div>
          <h1 className="text-2xl font-extrabold bg-clip-text text-transparent bg-gradient-to-r from-blue-600 to-pink-500 tracking-tight">
            EZ Cad AI
          </h1>
        </div>
        
        <input 
          type="file" 
          ref={fileInputRef} 
          onChange={handleFileUpload} 
          multiple 
          accept="application/pdf,image/png,image/jpeg,image/webp"
          className="hidden" 
        />
        
        <div className="flex flex-col gap-2 mb-6">
          <button 
            onClick={() => fileInputRef.current.click()}
            className="w-full bg-gradient-to-r from-blue-500 to-indigo-600 hover:from-blue-600 hover:to-indigo-700 text-white font-bold py-3.5 rounded-2xl shadow-md transition-all flex justify-center items-center gap-2 group"
          >
            <span className="text-xl group-hover:scale-110 transition-transform">+</span> Upload Drawing
          </button>

          <button 
            onClick={loadDemoDocument}
            className="w-full bg-white/60 hover:bg-white/90 border border-pink-200 text-pink-600 font-bold py-2 rounded-xl text-xs transition-all flex justify-center items-center gap-2"
          >
            ✨ Load Demo CAD Blueprint
          </button>
        </div>

        <h2 className="text-xs font-bold text-gray-500 uppercase tracking-widest mb-3 pl-1">Recent Documents</h2>
        
        <div className="flex flex-col gap-2 overflow-y-auto pr-1 pb-4 flex-1 custom-scrollbar">
          {documents.length === 0 ? (
            <div className="text-center text-sm text-gray-400 mt-4 italic bg-white/30 rounded-xl p-4 border border-dashed border-gray-300">
              No layouts uploaded yet.
            </div>
          ) : (
            documents.map((doc) => (
              <div 
                key={doc.id} 
                onClick={() => setSelectedDocId(doc.id)}
                className={`p-3 rounded-xl cursor-pointer transition-all flex items-center justify-between group border ${
                  selectedDocId === doc.id 
                    ? 'bg-blue-50/80 border-blue-300 shadow-sm' 
                    : 'bg-white/40 hover:bg-white/70 border-white/50'
                }`}
              >
                <div className="flex items-center gap-3 overflow-hidden">
                  <span className="text-blue-500 flex-shrink-0">
                    {doc.type === 'application/pdf' ? '📄' : '🖼️'}
                  </span>
                  <span className={`text-sm font-medium truncate ${selectedDocId === doc.id ? 'text-blue-700' : 'text-gray-700'}`}>
                    {doc.name}
                  </span>
                </div>
                <button 
                  onClick={(e) => removeDocument(e, doc.id)}
                  className="text-gray-400 hover:text-red-500 opacity-0 group-hover:opacity-100 transition-opacity"
                >
                  <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M6 18L18 6M6 6l12 12" /></svg>
                </button>
              </div>
            ))
          )}
        </div>
      </div>

      {/* --- CENTER: VIRTUAL CAD INTERACTIVE ENGINE (PDF & Image Viewer Fix) --- */}
      <div className="flex-1 my-4 bg-white/40 backdrop-blur-xl border border-white/60 rounded-3xl shadow-lg flex flex-col overflow-hidden relative z-0">
        
        {/* Upper Split (Document CAD Engine Viewport) */}
        <div className="flex-1 flex flex-col min-h-0 relative">
          {selectedDoc ? (
            <>
              {/* Toolbar */}
              <div className="w-full h-14 bg-white/50 border-b border-white/50 flex items-center px-4 justify-between backdrop-blur-md z-10">
                <span className="font-bold text-gray-700 truncate pr-4 text-sm">{selectedDoc.name}</span>
                
                {/* CAD Layers Controller */}
                <div className="flex items-center gap-1.5 flex-wrap">
                  <button 
                    onClick={() => setShowGrid(!showGrid)} 
                    className={`text-[10px] font-bold px-2.5 py-1 rounded-full transition-all border ${showGrid ? 'bg-blue-500/10 border-blue-500/30 text-blue-600' : 'bg-slate-100 border-slate-200 text-slate-500'}`}
                  >
                    Grid Line
                  </button>
                  <button 
                    onClick={() => setShowLoadVectors(!showLoadVectors)} 
                    className={`text-[10px] font-bold px-2.5 py-1 rounded-full transition-all border ${showLoadVectors ? 'bg-pink-500/10 border-pink-500/30 text-pink-600' : 'bg-slate-100 border-slate-200 text-slate-500'}`}
                  >
                    Load Vectors
                  </button>
                  <button 
                    onClick={() => setShowStressHeatmap(!showStressHeatmap)} 
                    className={`text-[10px] font-bold px-2.5 py-1 rounded-full transition-all border ${showStressHeatmap ? 'bg-indigo-500/10 border-indigo-500/30 text-indigo-600' : 'bg-slate-100 border-slate-200 text-slate-500'}`}
                  >
                    Stress Heatmap
                  </button>
                </div>
              </div>

              {/* Vector Rendering Board */}
              <div 
                ref={viewportRef}
                onMouseDown={handleMouseDown}
                onMouseMove={handleMouseMove}
                onMouseUp={handleMouseUp}
                onMouseLeave={handleMouseUp}
                className="flex-1 bg-slate-950 overflow-hidden relative cursor-grab active:cursor-grabbing flex items-center justify-center"
              >
                {/* CAD Blueprint Blueprint Layer */}
                <div 
                  className="absolute origin-center transition-transform duration-75"
                  style={{
                    transform: `translate(${pan.x}px, ${pan.y}px) scale(${zoom})`,
                    width: '600px',
                    height: '500px'
                  }}
                >
                  {/* Grid Lines Overlay */}
                  {showGrid && (
                    <div className="absolute inset-0 pointer-events-none" style={{
                      backgroundImage: 'radial-gradient(circle, rgba(255,255,255,0.07) 1px, transparent 1px)',
                      backgroundSize: '20px 20px'
                    }} />
                  )}

                  {/* Stress Heatmap layer */}
                  {showStressHeatmap && (
                    <div className="absolute inset-0 pointer-events-none mix-blend-screen bg-gradient-to-br from-indigo-500/20 via-pink-500/30 to-rose-600/40 animate-pulse duration-1000" />
                  )}

                  {/* SVG CAD Blueprint Elements */}
                  <svg className="w-full h-full" viewBox="0 0 600 500">
                    <rect width="600" height="500" fill="transparent" />

                    {/* Standard Blueprint Borders */}
                    <rect x="20" y="20" width="560" height="460" stroke="rgba(59, 130, 246, 0.4)" strokeWidth="2" fill="none" />
                    <rect x="25" y="25" width="550" height="450" stroke="rgba(59, 130, 246, 0.2)" strokeWidth="1" fill="none" />

                    {/* Check if Custom Uploaded Image is present */}
                    {!selectedDoc.isDemo && selectedDoc.type !== 'application/pdf' ? (
                      <image href={selectedDoc.dataUrl} x="50" y="50" width="500" height="400" preserveAspectRatio="xMidYMid meet" />
                    ) : (
                      // Render Interactive Mock Vector Layout (Ideal for raw PDFs or Demo Document)
                      <>
                        {/* Slab Boundary */}
                        <rect x="150" y="120" width="300" height="260" stroke="rgba(59, 130, 246, 0.6)" strokeWidth="3" fill="rgba(59,130,246,0.05)" strokeDasharray="5,5" />

                        {/* Beams */}
                        {DEMO_STRUCTURE.beams.map((beam, i) => (
                          <g key={`beam-${i}`}>
                            <line 
                              x1={beam.fromX} 
                              y1={beam.fromY} 
                              x2={beam.toX} 
                              y2={beam.toY} 
                              stroke="rgba(244, 63, 94, 0.8)" 
                              strokeWidth="4" 
                            />
                            <text 
                              x={(beam.fromX + beam.toX) / 2} 
                              y={((beam.fromY + beam.toY) / 2) - 8} 
                              fill="rgba(244, 63, 94, 0.9)" 
                              fontSize="10" 
                              fontWeight="bold" 
                              textAnchor="middle"
                            >
                              {beam.label}
                            </text>
                          </g>
                        ))}

                        {/* Columns */}
                        {DEMO_STRUCTURE.columns.map((col, i) => (
                          <g key={`col-${i}`}>
                            <rect 
                              x={col.x - 15} 
                              y={col.y - 15} 
                              width="30" 
                              height="30" 
                              fill="rgba(59, 130, 246, 0.8)" 
                              stroke="white" 
                              strokeWidth="1.5" 
                            />
                            <text 
                              x={col.x} 
                              y={col.y - 22} 
                              fill="#60a5fa" 
                              fontSize="10" 
                              fontWeight="bold" 
                              textAnchor="middle"
                            >
                              {col.label}
                            </text>
                            
                            {/* Load Vectors Visual Indicators */}
                            {showLoadVectors && (
                              <path 
                                d={`M ${col.x} ${col.y + 15} L ${col.x} ${col.y + 35} M ${col.x - 4} ${col.y + 30} L ${col.x} ${col.y + 35} L ${col.x + 4} ${col.y + 30}`} 
                                stroke="#f43f5e" 
                                strokeWidth="2" 
                                fill="none"
                              />
                            )}
                          </g>
                        ))}

                        {/* Dimensions & Spans */}
                        {DEMO_STRUCTURE.dimensions.map((dim, i) => (
                          <g key={`dim-${i}`}>
                            <line x1={dim.fromX} y1={dim.fromY} x2={dim.toX} y2={dim.toY} stroke="rgba(255,255,255,0.4)" strokeWidth="1" strokeDasharray="2,2" />
                            <text x={(dim.fromX + dim.toX) / 2} y={(dim.fromY + dim.toY) / 2 - 5} fill="rgba(255,255,255,0.6)" fontSize="9" textAnchor="middle">
                              {dim.label}
                            </text>
                          </g>
                        ))}

                        {/* Center Point Indicator */}
                        <circle cx="300" cy="250" r="4" fill="#a855f7" />
                        <text x="310" y="254" fill="#d8b4fe" fontSize="8" fontWeight="bold">CENTER PIER</text>
                      </>
                    )}

                    {/* General Drawing Sheet Info */}
                    <text x="40" y="440" fill="rgba(255,255,255,0.4)" fontSize="10">EZ CAD ENGINE v1.2</text>
                    <text x="40" y="455" fill="rgba(255,255,255,0.25)" fontSize="9">Coord: X: {pan.x} | Y: {pan.y}</text>
                    <text x="560" y="450" fill="#60a5fa" fontSize="12" fontWeight="bold" textAnchor="end">SHEET S102</text>
                  </svg>
                </div>

                {/* Onboard Floating Viewport Controllers */}
                <div className="absolute bottom-4 right-4 bg-black/60 backdrop-blur-md border border-white/10 rounded-xl p-2 flex gap-2 z-10">
                  <button onClick={() => handleZoom(1.2)} className="w-8 h-8 rounded-lg bg-white/10 hover:bg-white/20 text-white font-bold flex items-center justify-center text-sm" title="Zoom In">+</button>
                  <button onClick={() => handleZoom(0.8)} className="w-8 h-8 rounded-lg bg-white/10 hover:bg-white/20 text-white font-bold flex items-center justify-center text-sm" title="Zoom Out">-</button>
                  <button onClick={resetView} className="px-3 h-8 rounded-lg bg-white/10 hover:bg-white/20 text-white font-bold flex items-center justify-center text-xs" title="Reset Camera">Reset</button>
                </div>
              </div>
            </>
          ) : (
            <div className="flex-1 flex flex-col items-center justify-center bg-gray-50/30">
              <div className="w-24 h-24 mb-6 rounded-3xl bg-gradient-to-tr from-blue-100 to-pink-100 flex items-center justify-center shadow-inner border border-white">
                <span className="text-5xl">🏗️</span>
              </div>
              <h2 className="text-xl font-bold text-gray-700 mb-2">Workspace Empty</h2>
              <p className="text-gray-500 font-medium text-center max-w-xs text-sm px-4">
                Select or upload a structural drawing to preview it on our zoomable engineering vector canvas.
              </p>
            </div>
          )}
        </div>

        {/* --- LOWER SPLIT: SPECIALIZED LLM INSPECTION DRAWER --- */}
        <div className="h-72 border-t border-white/50 bg-white/60 backdrop-blur-xl flex flex-col z-10">
          <div className="bg-white/40 px-6 py-2 border-b border-white/40 flex justify-between items-center">
            <span className="text-xs font-bold text-slate-500 uppercase tracking-widest flex items-center gap-1.5">
              <span>🔧</span> AI Inspection Panel
            </span>
            <div className="flex gap-2">
              <button 
                onClick={() => setActiveInsightTab('audit')}
                className={`text-xs px-3 py-1 rounded-full font-bold transition-all ${activeInsightTab === 'audit' ? 'bg-blue-500 text-white' : 'hover:bg-white/50 text-slate-600'}`}
              >
                ✨ Audit Report
              </button>
              <button 
                onClick={() => setActiveInsightTab('boq')}
                className={`text-xs px-3 py-1 rounded-full font-bold transition-all ${activeInsightTab === 'boq' ? 'bg-pink-500 text-white' : 'hover:bg-white/50 text-slate-600'}`}
              >
                ✨ Materials (BoQ)
              </button>
              <button 
                onClick={() => setActiveInsightTab('rfi')}
                className={`text-xs px-3 py-1 rounded-full font-bold transition-all ${activeInsightTab === 'rfi' ? 'bg-indigo-500 text-white' : 'hover:bg-white/50 text-slate-600'}`}
              >
                ✨ Technical RFI
              </button>
            </div>
          </div>

          <div className="flex-1 p-5 flex flex-col md:flex-row gap-4 overflow-hidden">
            {/* Quick action runner panel */}
            <div className="w-full md:w-56 flex flex-col gap-2 justify-center border-b md:border-b-0 md:border-r border-white/40 pb-4 md:pb-0 md:pr-4">
              <p className="text-xs text-slate-500 leading-relaxed mb-1">
                Trigger Gemini 2.5 Flash to process the drawing with specialized engineering pipelines.
              </p>
              
              {activeInsightTab === 'audit' && (
                <button 
                  onClick={handleRunAudit}
                  disabled={isInsightLoading}
                  className="w-full bg-blue-500 hover:bg-blue-600 text-white text-xs font-bold py-2.5 rounded-xl shadow transition-all flex items-center justify-center gap-2 disabled:opacity-50"
                >
                  ✨ Analyze Load Paths
                </button>
              )}

              {activeInsightTab === 'boq' && (
                <button 
                  onClick={handleEstimateBoQ}
                  disabled={isInsightLoading}
                  className="w-full bg-pink-500 hover:bg-pink-600 text-white text-xs font-bold py-2.5 rounded-xl shadow transition-all flex items-center justify-center gap-2 disabled:opacity-50"
                >
                  ✨ Estimate Bill of Quantities
                </button>
              )}

              {activeInsightTab === 'rfi' && (
                <button 
                  onClick={handleDraftRFI}
                  disabled={isInsightLoading}
                  className="w-full bg-indigo-500 hover:bg-indigo-600 text-white text-xs font-bold py-2.5 rounded-xl shadow transition-all flex items-center justify-center gap-2 disabled:opacity-50"
                >
                  ✨ Draft Technical RFI
                </button>
              )}
            </div>

            {/* Insight Result Field */}
            <div className="flex-1 overflow-y-auto bg-white/40 border border-white/60 rounded-2xl p-4 text-xs leading-relaxed custom-scrollbar font-sans">
              {isInsightLoading ? (
                <div className="h-full flex flex-col items-center justify-center gap-3">
                  <div className="w-6 h-6 border-2 border-indigo-500 border-t-transparent rounded-full animate-spin"></div>
                  <span className="text-slate-500 font-semibold animate-pulse">Running Gemini 2.5 Flash analysis...</span>
                </div>
              ) : (
                <div className="whitespace-pre-wrap text-slate-700">
                  {activeInsightTab === 'audit' && insightOutput.audit}
                  {activeInsightTab === 'boq' && insightOutput.boq}
                  {activeInsightTab === 'rfi' && insightOutput.rfi}
                </div>
              )}
            </div>
          </div>
        </div>

      </div>

      {/* --- RIGHT SIDEBAR: CHATBOT INTERFACE --- */}
      <div className="w-96 m-4 ml-2 bg-white/40 backdrop-blur-xl border border-white/60 rounded-3xl flex flex-col shadow-lg overflow-hidden z-10">
        <div className="h-16 border-b border-white/50 flex items-center px-6 bg-white/30 backdrop-blur-md gap-3">
          <div className="relative">
            <div className="w-8 h-8 rounded-full bg-gradient-to-r from-blue-500 to-pink-500 flex items-center justify-center text-white font-bold shadow-sm">
              AI
            </div>
            <div className="absolute bottom-0 right-0 w-2.5 h-2.5 bg-green-400 border-2 border-white rounded-full"></div>
          </div>
          <div>
            <h2 className="font-bold text-gray-800 text-sm">EZ Assistant</h2>
            <p className="text-[10px] text-gray-500 uppercase tracking-wider font-semibold">Gemini 2.5 Flash</p>
          </div>
        </div>
        
        <div className="flex-1 p-5 overflow-y-auto flex flex-col gap-5 custom-scrollbar bg-white/10">
          {chatMessages.map((msg, idx) => (
            <div key={idx} className={`flex ${msg.role === 'user' ? 'justify-end' : 'justify-start'} animate-in fade-in slide-in-from-bottom-2`}>
              <div className={`p-3.5 rounded-2xl max-w-[85%] text-[14px] leading-relaxed shadow-sm ${
                msg.role === 'user' 
                  ? 'bg-gradient-to-br from-blue-500 to-blue-600 text-white rounded-br-sm' 
                  : 'bg-white/80 border border-white/60 text-gray-800 rounded-bl-sm backdrop-blur-sm'
              }`}>
                {msg.text.split('\n').map((line, i) => <p key={i} className={i !== 0 ? 'mt-2' : ''}>{line}</p>)}
              </div>
            </div>
          ))}
          
          {isTyping && (
            <div className="flex justify-start">
              <div className="bg-white/80 border border-white/60 p-4 rounded-2xl rounded-bl-sm shadow-sm backdrop-blur-sm flex gap-1.5 items-center">
                <div className="w-2 h-2 bg-blue-400 rounded-full animate-bounce" style={{ animationDelay: '0ms' }}></div>
                <div className="w-2 h-2 bg-pink-400 rounded-full animate-bounce" style={{ animationDelay: '150ms' }}></div>
                <div className="w-2 h-2 bg-blue-400 rounded-full animate-bounce" style={{ animationDelay: '300ms' }}></div>
              </div>
            </div>
          )}
          <div ref={chatEndRef} />
        </div>

        <div className="p-4 bg-white/30 border-t border-white/50 backdrop-blur-md">
          <div className="flex items-end gap-2 bg-white/70 border border-white/80 rounded-2xl p-1.5 shadow-inner transition-all focus-within:ring-2 focus-within:ring-blue-300">
            <textarea 
              value={chatInput}
              onChange={(e) => setChatInput(e.target.value)}
              onKeyDown={(e) => {
                if (e.key === 'Enter' && !e.shiftKey) {
                  e.preventDefault();
                  handleSendMessage();
                }
              }}
              placeholder="Ask about detail metrics, load margins, specs..." 
              className="flex-1 bg-transparent p-2.5 max-h-32 min-h-[44px] resize-none focus:outline-none text-sm text-gray-800 placeholder-gray-500"
              rows={1}
            />
            <button 
              onClick={handleSendMessage}
              disabled={isTyping || !chatInput.trim()}
              className="bg-gradient-to-r from-pink-500 to-pink-600 hover:from-pink-600 hover:to-pink-700 disabled:opacity-50 text-white p-2.5 rounded-xl transition-all shadow-md mb-0.5 flex-shrink-0"
            >
              <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M12 19l9 2-9-18-9 18 9-2zm0 0v-8" /></svg>
            </button>
          </div>
          <p className="text-center text-[10px] text-gray-400 mt-2">
            Workspace context synced with Gemini.
          </p>
        </div>
      </div>

      {/* Global Scrollbar Customization */}
      <style dangerouslySetInnerHTML={{__html: `
        .custom-scrollbar::-webkit-scrollbar {
          width: 6px;
        }
        .custom-scrollbar::-webkit-scrollbar-track {
          background: transparent;
        }
        .custom-scrollbar::-webkit-scrollbar-thumb {
          background-color: rgba(156, 163, 175, 0.3);
          border-radius: 10px;
        }
        .custom-scrollbar::-webkit-scrollbar-thumb:hover {
          background-color: rgba(156, 163, 175, 0.5);
        }
      `}} />
    </div>
  );
}
